package main

import (
	"fmt"
	"strings"
)

func main() {
    msgArr := []string{
        "H", "e", "l", "l", "o", 
        " ", 
        "W", "o", "r", "l", "d", "!"
    }
	msg := strings.Join(msgArr, "")

	fmt.Println(msg)
    // => Hello World!
}

(require '[clojure.string :as clo-str])

(def msg 
    (-> [
        "H" "e" "l" "l" "o" 
        " " 
        "W" "o" "r" "l" "d" "!"
        ]

(defn -main []
    (println msg))
;; => Hello World!
